import torch
import torch.nn as nn
from torch_geometric.loader import DataLoader
from torch_geometric.nn import GCNConv, global_mean_pool

from firmware_dataset_light import FirmwareLightDataset


class GNNFirmwareClassifier(nn.Module):
    """
    Simple GCN-based graph classifier for firmware graphs.

    Each graph = one firmware
    Node features: [entropy, size_norm, offset_norm, sus_flag] (4 dims)
    Output: 2 classes -> 0 (clean), 1 (tampered)
    """
    def __init__(self, in_feats=4, hidden_dim=32, num_classes=2):
        super().__init__()
        self.conv1 = GCNConv(in_feats, hidden_dim)
        self.conv2 = GCNConv(hidden_dim, hidden_dim)
        self.lin = nn.Linear(hidden_dim, num_classes)

    def forward(self, x, edge_index, batch):
        # x: [N, in_feats], edge_index: [2, E], batch: [N]
        x = self.conv1(x, edge_index)
        x = torch.relu(x)
        x = self.conv2(x, edge_index)
        x = torch.relu(x)

        # Aggregate node embeddings to graph-level embedding
        x = global_mean_pool(x, batch)  # [num_graphs, hidden_dim]

        out = self.lin(x)  # [num_graphs, num_classes]
        return out


def train_epoch(model, loader, optimizer, criterion, device):
    model.train()
    total_loss = 0.0

    for batch in loader:
        batch = batch.to(device)

        optimizer.zero_grad()
        out = model(batch.x, batch.edge_index, batch.batch)
        loss = criterion(out, batch.y)
        loss.backward()
        optimizer.step()

        total_loss += loss.item() * batch.num_graphs

    avg_loss = total_loss / len(loader.dataset)
    return avg_loss


@torch.no_grad()
def eval_epoch(model, loader, device):
    model.eval()
    correct = 0
    total = 0

    for batch in loader:
        batch = batch.to(device)
        out = model(batch.x, batch.edge_index, batch.batch)
        preds = out.argmax(dim=1)
        correct += (preds == batch.y).sum().item()
        total += batch.num_graphs

    acc = correct / total if total > 0 else 0.0
    return acc


def main():
    # 1) Hyperparameters
    num_graphs = 300          # total synthetic firmware graphs
    batch_size = 16
    hidden_dim = 32
    num_epochs = 30
    lr = 1e-3

    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    print(f"Using device: {device}")

    # 2) Dataset
    dataset = FirmwareLightDataset(num_graphs=num_graphs)

    # Simple train/test split: 80% / 20%
    train_size = int(0.8 * len(dataset))
    test_size = len(dataset) - train_size
    train_dataset = dataset[:train_size]
    test_dataset = dataset[train_size:]

    train_loader = DataLoader(train_dataset, batch_size=batch_size, shuffle=True)
    test_loader = DataLoader(test_dataset, batch_size=batch_size)

    # 3) Model, optimizer, loss
    model = GNNFirmwareClassifier(in_feats=4, hidden_dim=hidden_dim, num_classes=2).to(device)
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    criterion = nn.CrossEntropyLoss()

    # 4) Training loop
    for epoch in range(1, num_epochs + 1):
        train_loss = train_epoch(model, train_loader, optimizer, criterion, device)
        test_acc = eval_epoch(model, test_loader, device)

        print(f"Epoch {epoch:02d} | Train Loss: {train_loss:.4f} | Test Acc: {test_acc:.4f}")

    # 5) Save model
    torch.save(model.state_dict(), "gnn_firmware_light.pth")
    print("Model saved to gnn_firmware_light.pth")


if __name__ == "__main__":
    main()
